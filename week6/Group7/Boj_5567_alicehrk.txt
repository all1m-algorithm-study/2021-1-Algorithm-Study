#include iostream
using namespace std;

int n, m;
const int MAX = 501;
int map[MAX][MAX];
bool visited[MAX];
bool S_Friend[MAX]; 상근이와 친구 여부
int num = 0;

void marriage() {
    상근이의 친구 
    for (int i = 2; i = n; i++) {
        if (map[1][i] == 1) {
            visited[i] = true; 방문
           S_Friend[i] = true; 상근이와 친구
        }
    }
    상근이의 친구의 친구
    for (int i = 2; i = n; i++) {
        if (S_Friend[i]) { 상근이와 친구 
            for (int j = 1; j = n; j++) {
                if (map[i][j]) { 친구의 친구
                    visited[j] = true; 방문
                }
            }
        }
    }
    초대 인원 계산
    for (int i = 2; i = n; i++) {
        if (visited[i]) {
            num++;
        }
    }
}

int main() {
    cin  n;
    cin  m;

    while (m--) {
        int a, b;
        cin  a  b;
        map[a][b] = 1;
        map[b][a] = 1;
    }

    marriage();
    cout  num;
}
